# teacher-paper skill 依赖（一次性安装即可在任何电脑使用）
# 安装：pip3 install -r requirements.txt
python-docx              # 生成 Word 试卷与答案（make_paper.py，必装）；也用于读 .docx 素材
pdfplumber               # 读取 .pdf 素材（含表格）；可选，未装时回退 pypdf
python-pptx              # 读取 .pptx 课件素材
openpyxl                 # 读取 .xlsx 表格素材
readability-lxml         # 网页正文提取（fetch_web.py 第2策略）；可选
rapidocr-onnxruntime     # 图片OCR兜底（ocr_image.py）；仅纯文本模型/无多模态识图时才需，离线可用
