openpyxl                 # 读取 .xlsx 表格素材
readability-lxml         # 网页正文提取（fetch_web.py 第2策略）；可选
rapidocr-onnxruntime     # 图片OCR兜底（ocr_image.py）；仅纯文本模型/无多模态识图时才需，离线可用
